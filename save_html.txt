 def save_html(self, intervals, rates):
        with open('exchange_report.html', 'w') as file_object:
            file_object.write("<html><head> </head> ")
            
            file_object.write("<table>")
            
            file_object.write("<thead>")
            
            file_object.write("<th>")
            
            file_object.write("Intervals")
            
            file_object.write("</th>")
            
                
            file_object.write("<th>")
            
            file_object.write("Rates")
            
            file_object.write("</th>")
            
            
            
            
            file_object.write("</thead>")
            
            
            for rate in rates:
                file_object.write(f"<tr><td> {rate}</td> </tr>")
            
            for interval in intervals:
                 file_object.write(f" <tr><td> {interval}</td> </tr>")
                
            file_object.write("</table>")
            
            file_object.write("</body></html>")
